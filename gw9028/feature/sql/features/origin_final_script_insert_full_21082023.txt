	TO_CHAR(TO_DATE('{RPT_DT}','DD-MM-YY'), 'DD-MM-YYYY') AS RPT_DT,
    CURRENT_TIMESTAMP ADD_TSTP
FROM
    (SELECT CUSTOMER_CDE,
        NVL(MONTHS_BETWEEN(LEAD(TXN_DT) OVER (PARTITION BY CUSTOMER_CDE ORDER BY TXN_DT), TXN_DT), MONTHS_BETWEEN(TO_DATE('{RPT_DT}','DD-MM-YY'), TXN_DT)) TXN_GAP
    FROM DW_ANALYTICS.DW_EWALL_TRANSACTION_FCT

    AND TXN_STATUS = 'S'
    AND CUSTOMER_CDE IN
        (SELECT CUSTOMER_CDE 
        FROM CINS_TMP_CUST_TEST)
    UNION ALL
    SELECT B.CUSTOMER_CDE,
        MONTHS_BETWEEN(MIN(TXN_DT), MAX(IDENTIFY_DT)) TXN_GAP
    FROM DW_ANALYTICS.DW_EWALL_TRANSACTION_FCT A
    JOIN DW_ANALYTICS.DW_EWALL_USER_DIM B
    ON A.CUSTOMER_CDE = B.CUSTOMER_CDE
    WHERE TXN_DT < TO_DATE('{RPT_DT}','DD-MM-YY')
    AND TXN_STATUS = 'S'
    AND TXN_DT > IDENTIFY_DT
    AND B.CUSTOMER_CDE IN
        (SELECT CUSTOMER_CDE 
        FROM CINS_TMP_CUST_TEST)
    GROUP BY B.CUSTOMER_CDE)
WHERE TXN_GAP > 12 AND TXN_GAP <= 36
GROUP BY CUSTOMER_CDE;
